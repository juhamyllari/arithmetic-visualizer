3   /**
4    * An ActivationPattern indicates which elements of each array are active.
5    * Activated elements are highlighted in the visualization animation frame.
6    * They correspond to the array elements which contribute to that suboperation.
7    */
8   public class ActivationPattern {
9       
10      public static final ActivationPattern ADDITION = new ActivationPattern("element", "element", "element");
11      public static final ActivationPattern SUBTRACTION = ADDITION;
12      public static final ActivationPattern LEFTSCALARMULTIPLICATION = new ActivationPattern("all", "element", "element");
13      public static final ActivationPattern RIGHTSCALARMULTIPLICATION = new ActivationPattern("element", "all", "element");
14      public static final ActivationPattern MATRIXMULTIPLICATION = new ActivationPattern("row", "column", "element");
15      
16      String leftPattern;
17      String rightPattern;
18      String resultPattern;
19  
20      public ActivationPattern(String leftPattern, String rightPattern, String resultPattern) {
21          this.leftPattern = leftPattern;
22          this.rightPattern = rightPattern;
23          this.resultPattern = resultPattern;
24      }
25  
26      public String getLeftPattern() {
27          return leftPattern;
28      }
29  
30      public String getRightPattern() {
31          return rightPattern;
32      }
33  
34      public String getResultPattern() {
35          return resultPattern;
36      }
37  
38  }
