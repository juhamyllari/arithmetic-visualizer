3   /**
4    * A SubOperation object represents a single frame in a visualization
5    * animation. It contains BooleanMasks to indicate active elements
6    * in each array, a BooleanMask to indicate which elements of the
7    * result array are to be displayed, and a String representation of
8    * the suboperation (e.g. a single dot product in the case of matrix
9    * multiplication).
10   */
11  public class SubOperation {
12      
13      private final BooleanMask leftActivation;
14      private final BooleanMask rightActivation;
15      private final BooleanMask resultActivation;
16      private final BooleanMask showResult;
17      private final String subOpString;
18  
19      public SubOperation(BooleanMask leftActivation, BooleanMask rightActivation, BooleanMask resultActivation, BooleanMask showResult, String subOpString) {
20          int m = leftActivation.getMask().length;
21          int n = leftActivation.getMask()[0].length;
22          this.leftActivation = leftActivation;
23          this.rightActivation = rightActivation;
24          this.resultActivation = resultActivation;
25          this.showResult = showResult;
26          this.subOpString = subOpString;
27      }
28  
29      public BooleanMask getLeftActivation() {
30          return leftActivation;
31      }
32  
33      public BooleanMask getRightActivation() {
34          return rightActivation;
35      }
36  
37      public BooleanMask getResultActivation() {
38          return resultActivation;
39      }
40  
41      public BooleanMask getShow() {
42          return showResult;
43      }
44  
45      public String getSubOpString() {
46          return subOpString;
47      }
48   
49  }
