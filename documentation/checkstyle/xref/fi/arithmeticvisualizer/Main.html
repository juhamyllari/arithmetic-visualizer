1   package fi.arithmeticvisualizer;
2   
3   import java.io.IOException;
4   import javafx.application.Application;
5   import static javafx.application.Application.launch;
6   import javafx.application.Platform;
7   import javafx.fxml.FXMLLoader;
8   import javafx.scene.Parent;
9   import javafx.scene.Scene;
10  import javafx.scene.control.Alert;
11  import javafx.scene.control.ButtonType;
12  import javafx.stage.Stage;
13  
14  /**
15   * The main class of the Arithmetic Visualizer application. Loads the entry
16   * scene of the application. For the controller class of the entry scene, see
17   * EntrySceneController.
18   */
19  public class Main extends Application {
20  
21      @Override
22      public void start(Stage stage) {
23  
24          try {
25              Parent root = FXMLLoader.load(getClass().getResource("/fxml/EntryScene.fxml"));
26              Scene scene = new Scene(root);
27              stage.setTitle("Arithmetic Visualizer");
28              stage.setScene(scene);
29          } catch (IOException ex) {
30              exitOnFailureToLoadScene("entry");
31          }
32          stage.show();
33      }
34  
35      /**
36       * The main() method is ignored in correctly deployed JavaFX application.
37       * main() serves only as fallback in case the application can not be
38       * launched through deployment artifacts, e.g., in IDEs with limited FX
39       * support. NetBeans ignores main().
40       *
41       * @param args the command line arguments
42       */
43      public static void main(String[] args) {
44          launch(args);
45      }
46  
47      /**
48       * Alerts the user that the program will terminate due to failure to load
49       * the next scene and exits.
50       *
51       * @param sceneName name of the scene which failed to load
52       */
53      public static void exitOnFailureToLoadScene(String sceneName) {
54          Alert alert = new Alert(Alert.AlertType.ERROR,
55                  "Unable to load " +  sceneName + " scene. Exiting.");
56          alert.showAndWait()
57                  .filter(response -> response == ButtonType.OK)
58                  .ifPresent(response -> Platform.exit());
59      }
60  
61  }
