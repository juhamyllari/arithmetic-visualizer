13  /**
14   * This is the controller class for the second (evaluation) scene of the application.
15   * The evaluation scene is where expressions are visualized.
16   */
17  public class EvaluationSceneController {
18  
19      private BinaryNode node;
20      private Visualizer visualizer;
21      private GraphicArray left;
22      private GraphicArray right;
23      private GraphicArray result;
24  
25      @FXML
26      private Button startButton;
27  
28      @FXML
29      private GridPane leftGrid;
30  
31      @FXML
32      private GridPane rightGrid;
33  
34      @FXML
35      private GridPane resultGrid;
36  
37      @FXML
38      private Text symbol;
39  
40      @FXML
41      private Text subOpText;
42  
43      @FXML
44      private HBox lowerRow;
45  
46      @FXML
47      private GridPane entryOptionsGrid;
48  
49      @FXML
50      private ChoiceBox operandChoiceBox;
51  
52      public void initData(BinaryNode node) {
53          this.node = node;
54          this.left = new GraphicArray(leftGrid, node.getLeft().evaluate());
55          this.right = new GraphicArray(rightGrid, node.getRight().evaluate());
56          this.result = new GraphicArray(resultGrid, node.evaluate());
57          
58          setOptionsGridVisibility(false);
59          this.symbol.setText(node.getSymbol());
60          this.visualizer = new Visualizer(this, node, left, right, result, subOpText);
61          visualizer.drawOperands();
62      }
63  
64      @FXML
65      public void initialize() {
66  
67      }
68  
69      @FXML
70      private void startButtonPush() {
71          startButton.setText("Restart visualization");
72          visualizer.visualize();
73      }
74  
75      public void setOptionsGridVisibility(boolean visible) {
76          this.entryOptionsGrid.setVisible(visible);
77      }
78  
79  }
